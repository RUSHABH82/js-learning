1. Ternary operator
Syntax:  condition ? trueValue : falseValue

    function testTernary(age) {
        let value = age === undefined ? 18 : age;
        console.log(value)
    }
2. Guard operator
Syntax:  value || defaultValue

    function testGuard(age) {
        let value = age || 18
        console.log(value)
    }
3. Default operator
Syntax:  value || fallbackValue

    function testDefault(age) {
        let value = age ?? 18
        console.log(value)
    }
